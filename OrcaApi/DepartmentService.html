def list(base_date = "")
  api_path = "/api01rv2/system01lstv2"
  req_name = "system01_managereq"

  body = {
    req_name => {
      "Request_Number" => "01",
      "Base_Date" => base_date,
    }
  }
  Result.new(orca_api.call(api_path, body: body))
end
