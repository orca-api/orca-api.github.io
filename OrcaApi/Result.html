def self.trim_response(hash)
  result = {}
  hash.each do |k, v|
    result[k] = case v
                when Hash
                  trim_response(v)
                when Array
                  v.reverse_each.drop_while(&:empty?).reverse.map { |e| trim_response(e) }
                else
                  v
                end
  end
  result
end
