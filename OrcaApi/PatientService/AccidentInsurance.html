def update(id, params)
  res = call_01(id)
  if !res.locked?
    locked_result = res
  end
  if !res.ok?
    return res
  end

  res = call_02(params, res)
  if !res.ok?
    return res
  end

  res = call_03(res)
  if res.ok?
    locked_result = nil
  end
  res
ensure
  unlock(locked_result)
end
