def get(id, pi_id, number, start_date)
  res = call_01(id)
  if !res.ok?
    return res
  end

  res = call_02(pi_id, res)
  if !res.ok?
    return res
  end

  res = call_04(number, start_date, res)
  res
ensure
  unlock(res)
end

def update(id, pi_id, number, start_date, args)
  locked_result = res = call_01(id)
  if !res.ok?
    return res
  end

  res = call_02(pi_id, res)
  if !res.ok?
    return res
  end

  res = call_04(number, start_date, res)
  if !res.ok?
    return res
  end

  res = call_05(args, res)
  if res.ok?
    locked_result = nil
  end
  res
ensure
  unlock(locked_result)
end
